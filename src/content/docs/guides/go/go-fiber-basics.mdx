Fiber is an Express-inspired web framework built on top of Fasthttp, the fastest HTTP engine for Go. This guide helps you create your first API.

## Prerequisites

- Go installed on your system
- Basic Go knowledge

```bash
# Verify Go installation
go version
```

## Create a New Project

<Steps>

1. **Create project folder**

   ```bash
   mkdir my-fiber-api
   cd my-fiber-api
   ```

2. **Initialize Go module**

   ```bash
   go mod init my-fiber-api
   ```

   This creates a `go.mod` file for dependency management.

3. **Install Fiber**

   ```bash
   go get github.com/gofiber/fiber/v2
   ```

</Steps>

## Basic Routing

```go
package main

import "github.com/gofiber/fiber/v2"

func main() {
    app := fiber.New()

    // GET request
    app.Get("/", func(c *fiber.Ctx) error {
        return c.SendString("GET request")
    })

    // POST request
    app.Post("/users", func(c *fiber.Ctx) error {
        return c.SendString("POST request - Create user")
    })

    // PUT request
    app.Put("/users/:id", func(c *fiber.Ctx) error {
        id := c.Params("id")
        return c.SendString("PUT request - Update user " + id)
    })

    // DELETE request
    app.Delete("/users/:id", func(c *fiber.Ctx) error {
        id := c.Params("id")
        return c.SendString("DELETE request - Delete user " + id)
    })

    app.Listen(":3000")
}
```

## Route Parameters

```go
// Path parameter
app.Get("/users/:id", func(c *fiber.Ctx) error {
    id := c.Params("id")
    return c.SendString("User ID: " + id)
})

// Optional parameter
app.Get("/users/:id?", func(c *fiber.Ctx) error {
    id := c.Params("id", "default")
    return c.SendString("User ID: " + id)
})

// Query parameters
app.Get("/search", func(c *fiber.Ctx) error {
    query := c.Query("q")
    page := c.Query("page", "1")
    return c.SendString("Search: " + query + ", Page: " + page)
})
```

app.Get("/api/user", func(c *fiber.Ctx) error {

### With Struct

```go
type User struct {
    Name  string `json:"name"`
    Email string `json:"email"`
    Age   int    `json:"age"`
}

app.Get("/api/user", func(c *fiber.Ctx) error {
    user := User{
        Name:  "John Doe",
        Email: "john@example.com",
        Age:   30,
    }
    return c.JSON(user)
})
```

## Request Body

```go
type CreateUserRequest struct {
    Name  string `json:"name"`
    Email string `json:"email"`
}

app.Post("/users", func(c *fiber.Ctx) error {
    req := new(CreateUserRequest)

    if err := c.BodyParser(req); err != nil {
        return c.Status(400).JSON(fiber.Map{
            "error": "Invalid request body",
        })
    }

    return c.JSON(fiber.Map{
        "message": "User created",
        "user":    req,
    })
})
```

## Middleware

```go
// Logger middleware
app.Use(logger.New())

// Custom middleware
app.Use(func(c *fiber.Ctx) error {
    fmt.Println("Request received:", c.Path())
    return c.Next()
})

// Route-specific middleware
app.Get("/admin", AuthMiddleware, func(c *fiber.Ctx) error {
    return c.SendString("Admin area")
})

func AuthMiddleware(c *fiber.Ctx) error {
    token := c.Get("Authorization")
    if token == "" {
        return c.Status(401).JSON(fiber.Map{
            "error": "Unauthorized",
        })
    }
    return c.Next()
}
```

---

## Error Handling

```go
app.Use(func(c *fiber.Ctx) error {
    err := c.Next()

    if err != nil {
        code := fiber.StatusInternalServerError
        if e, ok := err.(*fiber.Error); ok {
            code = e.Code
        }

        return c.Status(code).JSON(fiber.Map{
            "error": err.Error(),
        })
    }

    return nil
})
```
